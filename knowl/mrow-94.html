\begin{align*} \dfrac{d}{dx}(x^3 + 3y^3) \amp = \dfrac{d}{dx}(3xy)\\ \dfrac{d}{dx}(x^3) + \dfrac{d}{dx}(3y^3) \amp = 3x \cdot \dfrac{dy}{dx} + y \cdot \dfrac{d}{dx}(3x) \amp \text{(separate the terms)}\\ 3x^2 + 3 \cdot 3y^2 \dfrac{dy}{dx} \amp = 3x \cdot \dfrac{dy}{dx} + y \cdot 3 \amp \text{(apply the chain rule)}\\ 3x^2 + 9y^2 \dfrac{dy}{dx} \amp = 3x \dfrac{dy}{dx} + 3y \amp \text{(solve for }\dfrac{dy}{dx})\\ 9y^2 \dfrac{dy}{dx} - 3x\dfrac{dy}{dx} \amp = 3y - 3x\\ \dfrac{dy}{dx}(9y^2 -3x) \amp = 3y - 3x\\ \dfrac{dy}{dx} \amp = \dfrac{3y - 3x}{9y^2 -3x} \end{align*}
