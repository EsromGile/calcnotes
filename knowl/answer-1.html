\begin{align*} \int x \cos(x) dx \amp = x \sin(x) - \int \sin(x) \cdot 1 dx\\ \amp = x \sin(x) - ( - \cos(x)) + c \\ \amp = x \sin(x) + \cos(x) + c \end{align*}
\begin{equation*} \begin{cases} \text{}\\ u = x \\ v = \sin(x) \\ du = 1 dx \\ dv = \cos(x) dx \end{cases} \end{equation*}
